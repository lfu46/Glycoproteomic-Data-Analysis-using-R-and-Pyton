Iron_GO_GSEA <- gseGO(
  geneList = genelist,
  # 'ALL' for all 'BP', 'MF', and 'CC'

  # https://bioconductor.org/packages/release/data/annotation/manuals/org.Hs.eg.db/man/org.Hs.eg.db.pdf
